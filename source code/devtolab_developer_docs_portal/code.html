import { DevToLab } from '@devtolab/sdk';

// Initialize with your API key
const client = new DevToLab({
  apiKey: 'dtl_test_51...xyz'
});

async function main() {
  const response = await client.projects.create({
    name: 'HyperScale-Backend',
    region: 'us-east-1'
  });
  
  console.log('Project ID:', response.id);
}

main();
